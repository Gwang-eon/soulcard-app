# 완전한 타로카드 해석 데이터베이스

> **목적**: 즉시 사용 가능한 78장 타로카드 완전 해석 데이터  
> **구성**: 78장 × 6카테고리 × 2방향 = 936개 해석 + 300개 조합  
> **활용**: 프로덕션 레벨 AI 타로 앱 직접 구현 가능

---

# 📚 데이터베이스 구조

```typescript
interface CompleteTarotCard {
  id: number;
  name: string;
  koreanName: string;
  suit: 'major' | 'wands' | 'cups' | 'swords' | 'pentacles';
  number?: number;
  element?: 'fire' | 'water' | 'air' | 'earth';
  
  // 핵심 키워드 (각 방향별 5-7개)
  uprightKeywords: string[];
  reversedKeywords: string[];
  
  // 6개 카테고리별 상세 해석 (각 200-300자)
  interpretations: {
    upright: DetailedMeanings;
    reversed: DetailedMeanings;
  };
  
  // 3가지 타임프레임별 해석
  timing: {
    immediate: string;    // 즉시-1주
    shortTerm: string;    // 1주-3개월  
    longTerm: string;     // 3개월-1년
  };
  
  // 심리학적 해석
  psychology: {
    upright: string;
    reversed: string;
    jungian: string;      // 융 심리학적 해석
    chakra?: string;      // 차크라 연결 (해당시)
  };
  
  // 실용적 조언 (액션 아이템)
  practicalAdvice: {
    upright: ActionAdvice;
    reversed: ActionAdvice;
  };
  
  // 다른 카드와의 시너지/갈등
  cardRelations: {
    synergy: number[];    // 시너지 카드 ID들
    conflict: number[];   // 갈등 카드 ID들
    amplifies: number[];  // 증폭시키는 카드들
  };
}

interface DetailedMeanings {
  general: string;
  love: string;
  career: string;
  money: string;
  health: string;
  spiritual: string;
}

interface ActionAdvice {
  whatToDo: string;      // 구체적 행동 지침
  whatToAvoid: string;   // 피해야 할 것들
  timeframe: string;     // 언제까지
  priority: 'high' | 'medium' | 'low';
}
```
